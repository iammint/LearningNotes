# Closure 闭包
```js
const myAlert = () => {
  const x = `Help! I think I found a clue!`
  let count = 0
  const alerter = () => {
    alert(`${x} ${++count}`)
  }
  return alerter
}
const funcAlert = myAlert()
const funcAlert2 = myAlert()
funcAlert()//1
funcAlert()//2
funcAlert2()//1
funcAlert()()//3
```


```js
const newClue = (name) => {
  const length = name.length
  return (weapon) => {
    let clue = length + weapon.length
    return !!(clue % 1)
  }
}
const didGreenDoItWithA = newClue("Green")
didGreenDoItWithA("candlestick")
didGreenDoItWithA("lead pipe")
//false
//fasle
```


```js
function countClues() {
  let n = 0
  return {
    count: () => { return ++n },
    reset: () => { return n = 0 }
  }
}
let counter = countClues()
// counter()//counter is not a function
counter.count()//1
counter.count()//2
counter.count()//3
counter.reset()//0
```


```js
const makeTimer = () => {
  let elapsed = 0
  const stopwatch = () => elapsed
  const increase = () => elapsed++
  setInterval(increase, 1000)
  return stopwatch
}
let timer = makeTimer()
timer()//0
//After x seconds, ivocate timer again
timer()//x
```
